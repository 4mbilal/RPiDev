Raspberry Pi does not connect to KAU wifi easily. Do the following

sudo nano /etc/wpa_supplicant/wpa_supplicant.conf
network={
      ssid="KAU-INTERNET"
      priority=1
      proto=RSN
      key_mgmt=WPA2-Enterprise
      pairwise=CCMP
      auth_alg=OPEN
      eap=PEAP
      identity="00062041"
      password="abcd"
      phase1="peaplabel=0"
      phase2="auth=MSCHAPV2"
      }
